USGS 過去 30 日 M3.0+ 地震データ取得 # ========================================================== url = "https://earthquake.usgs.gov/earthquakes/feed/v1.0/summary/3.0_month.csv" df = pd.read_csv(url) # ========================================================== # 2. 整形（日本時間） # ========================================================== df["time"] = pd.to_datetime(df["time"]) # UTC df["time_6h"] = df["time"].dt.floor("6H") df["lat"] = df["latitude"] df["lon"] = df["longitude"] df["mag"] = df["mag"].fillna(0) df["depth"] = df["depth"].fillna(0) # ========================================================== # 3. 色（浅・中・深） # ========================================================== def depth_color(d): if d < 70: return "#ff0000" # 浅い elif d < 300: return "#ffaa00" # 中間 else: return "#0000ff" # 深い df["color"] = df["depth"].apply(depth_color) df["radius"] = df["mag"].apply(lambda m: 2 + m * 2) # ========================================================== # 4. GeoJSON 生成（★ JST 表示は HTML 側でやる） # ========================================================== features = [] for _, r in df.iterrows(): features.append({ "type": "Feature", "geometry": { "type": "Point", "coordinates": [float(r["lon"]), float(r["lat"])] }, "properties": { "times": [r["time_6h"].isoformat()], "mag": float(r["mag"]), "depth": float(r["depth"]), "style": { "color": r["color"], "fillColor": r["color"], "fillOpacity": 0.7, "radius": r["radius"], "weight": 1 }, "icon": "circle" } }) eq_json = { "type": "FeatureCollection", "features": features } # 安全に JSON を JS に埋め込む eq_json_safe = json.dumps(eq_json).replace(" Earthquake Animation (JST)
