import folium from folium.plugins import TimestampedGeoJson import pandas as pd import json # ========================================================== # 1. USGS 過去30日・M3.0以上のデータ取得 # ========================================================== url = "https://earthquake.usgs.gov/earthquakes/feed/v1.0/summary/all_month.csv" df = pd.read_csv(url) df["time"] = pd.to_datetime(df["time"], utc=True) df = df[df["mag"] >= 3.0].reset_index(drop=True) df["lat"] = df["latitude"] df["lon"] = df["longitude"] df["depth"] = df["depth"].fillna(0) df["mag"] = df["mag"].fillna(0) # 6時間刻みに丸め（TimeDimension） df["time_6h"] = df["time"].dt.floor("6H") # ========================================================== # 2. Z (UTC) 形式へ強制変換（これが最重要） # ========================================================== def to_td_time(t): return t.strftime("%Y-%m-%dT%H:%M:%SZ") # ========================================================== # 3. 色とサイズ # ========================================================== def depth_color(d): if d < 70: return "#ff0000" # 赤 elif d < 300: return "#ffaa00" # 黄 else: return "#0000ff" # 青 df["color"] = df["depth"].apply(depth_color) df["radius"] = df["mag"].apply(lambda m: 2 + m * 2) # ========================================================== # 4. GeoJSON 生成 # ========================================================== features = [] for _, r in df.iterrows(): features.append({ "type": "Feature", "geometry": { "type": "Point", "coordinates": [r["lon"], r["lat"]], }, "properties": { "times": [to_td_time(r["time_6h"])], "mag": float(r["mag"]), "depth": float(r["depth"]), "style": { "color": r["color"], "fillColor": r["color"], "radius": r["radius"], "fillOpacity": 0.7, "weight": 1 }, "icon": "circle" } }) eq_json = json.dumps({ "type": "FeatureCollection", "features": features }) # ========================================================== # 5. HTML テンプレートに埋め込み（あなたのHTML） # ========================================================== html_template = """ Earthquake Animation (JST + AutoPlay + Plates)
深さ（Depth）
● 浅い（0–70km）
● 中間（70–300km）
● 深い（300km+）
""" # GeoJSON 埋め込み html_output = html_template.replace("__EQ_JSON__", eq_json) with open("index.html", "w", encoding="utf-8") as f: f.write(html_output) print("✔ index.html を生成しました！ GitHub Pages に置けば動きます。")
